```rust
fn format_duration(ms: u64) -> String {
    if ms < 1000 {
        format!("{}ms", ms)
    } else if ms < 60_000 {
        format!("{:.1}s", ms as f64 / 1000.0)
    } else {
        let mins = ms / 60_000;
        let secs = (ms % 60_000) / 1000;
        format!("{}m {}s", mins, secs)
    }
}

/// Check if terminal supports Unicode
pub fn supports_unicode() -> bool {
    // Simple heuristic - check TERM and locale
    std::env::var("TERM").map(|t| !t.contains("dumb")).unwrap_or(true)
        && std::env::var("LANG")
            .map(|l| l.contains("UTF") || l.contains("utf"))
            .unwrap_or(true)
}

/// Check if terminal supports colors
pub fn supports_color() -> bool {
    console::colors_enabled_stderr()
```

```rust
//! Chibby CLI - Local-first CI/CD
//!
//! A standalone CLI that shares the same engine as the desktop app.

use clap::{Parser, Subcommand};
use owo_colors::OwoColorize;
use std::path::PathBuf;

mod cli_commands;

// Import the styled printer
#[path = "../cli/mod.rs"]
mod cli;

use cli::{icons, Printer, StageStatus};

#[derive(Parser)]
#[command(
    name = "chibby",
    about = "Local-first CI/CD for solo developers",
    version,
    after_help = "Run 'chibby <command> --help' for more information on a command."
)]
struct Cli {
    #[command(subcommand)]
    command: Option<Commands>,

    /// Enable verbose output
    #[arg(short, long, global = true)]
    verbose: bool,

    /// Disable colors and emoji
    #[arg(long, global = true)]
    no_color: bool,
}

#[derive(Subcommand)]
enum Commands {
    /// Run the pipeline for the current project
    Run {
        /// Environment to deploy to (e.g., staging, production)
        #[arg(short, long)]
        env: Option<String>,

        /// Only run specific stages
        #[arg(short, long)]
        stage: Vec<String>,

        /// Project path (defaults to current directory)
        #[arg(short, long)]
        project: Option<PathBuf>,

        /// Skip preflight checks
        #[arg(long)]
        skip_preflight: bool,
    },

    /// Show status of the current or last run
    Status {
        /// Project path
        #[arg(short, long)]
        project: Option<PathBuf>,
    },

    /// Cancel a running pipeline
    Cancel {
        /// Project path
        #[arg(short, long)]
        project: Option<PathBuf>,
    },

    /// Manage projects
    #[command(subcommand)]
    Projects(ProjectsCmd),

    /// Manage pipelines
    #[command(subcommand)]
    Pipeline(PipelineCmd),

    /// View run history
    History {
        /// Project path
        #[arg(short, long)]
        project: Option<PathBuf>,

        /// Filter by environment
        #[arg(short, long)]
        env: Option<String>,

        /// Number of runs to show
        #[arg(short, long, default_value = "10")]
        limit: usize,
    },

    /// Retry a failed run
    Retry {
        /// Run ID to retry
        run_id: String,

        /// Start from a specific stage
        #[arg(long)]
        from_stage: Option<String>,
    },

    /// Rollback to a previous successful run
    Rollback {
        /// Run ID to rollback to
        run_id: String,
    },

    /// Manage environment variables and secrets
    #[command(subcommand)]
    Secrets(SecretsCmd),

    /// Manage environments
    #[command(subcommand)]
    Env(EnvCmd),

    /// Version management
    #[command(subcommand)]
    Version(VersionCmd),

    /// Artifact management
    #[command(subcommand)]
    Artifact(ArtifactCmd),

    /// Security and quality scans
    #[command(subcommand)]
    Scan(ScanCmd),

    /// Run preflight checks

#[tokio::main]
async fn main() {
    let cli = Cli::parse();

    // Handle color preferences
    if cli.no_color {
        owo_colors::set_override(false);
    }

    let printer = Printer::new(cli.verbose);

    match &cli.command {
        None => {
            // No command - show banner and help hint
            printer.banner();
            println!(
                "  {} {}",
                "Type".bright_black(),
                "chibby --help".cyan().bold()
            );
            println!(
                "  {} {}",
                "or".bright_black(),
                "chibby run".cyan().bold()
            );
            println!();
        }

        Some(Commands::Run { env, stage, project, skip_preflight }) => {
            run_pipeline(&printer, env.as_deref(), stage, project.as_ref(), *skip_preflight).await;
        }

        Some(Commands::Status { project }) => {
            show_status(&printer, project.as_ref()).await;
        }

        Some(Commands::Projects(cmd)) => {
            handle_projects(&printer, cmd).await;
        }

        Some(Commands::Pipeline(cmd)) => {
            handle_pipeline(&printer, cmd).await;
        }

        Some(Commands::History { project, env, limit }) => {
            show_history(&printer, project.as_ref(), env.as_deref(), *limit).await;
        }

        Some(Commands::Init { path }) => {
            init_project(&printer, path.as_ref()).await;
        }

        Some(Commands::Preflight { env, project }) => {
            run_preflight(&printer, env.as_deref(), project.as_ref()).await;
        }

        Some(Commands::App) => {
            open_app(&printer);
        }

        // ... handle other commands similarly
        _ => {
            printer.warn("This command is not yet implemented");
        }
    }
}

async fn run_pipeline(
    printer: &Printer,
    env: Option<&str>,
    stages: &[String],
    project: Option<&PathBuf>,
    skip_preflight: bool,
) {
    printer.banner();
    
    let project_path = project
        .cloned()
        .unwrap_or_else(|| std::env::current_dir().unwrap());

    printer.header("Running Pipeline");
    printer.kv("Project", &project_path.display().to_string());
    if let Some(e) = env {
        printer.kv("Environment", e);
    }
    if !stages.is_empty() {
        printer.kv("Stages", &stages.join(", "));
    }
    printer.newline();

    // Preflight
    if !skip_preflight {
        let spin = cli::spinner("Running preflight checks...");
        // TODO: Actually run preflight
        std::thread::sleep(std::time::Duration::from_millis(500));
        spin.finish_and_clear();
        printer.success("Preflight checks passed");
        printer.newline();
    }

    // Simulate pipeline stages
    let example_stages = ["build", "test", "deploy"];
    
    for (i, stage) in example_stages.iter().enumerate() {
        printer.stage(stage, StageStatus::Running);
        
        // Simulate work
        let spin = cli::spinner(&format!("Running {}...", stage));
        std::thread::sleep(std::time::Duration::from_millis(800));
        spin.finish_and_clear();

        if i == 1 && stages.contains(&"fail".to_string()) {
            // Simulated failure
            printer.stage(stage, StageStatus::Failed);
            printer.log("error", "Command exited with code 1");
            printer.run_summary("failed", 1234, i, example_stages.len());
            return;
        }

        printer.stage(stage, StageStatus::Success);
    }

    printer.run_summary("success", 2345, example_stages.len(), example_stages.len());
}

async fn run_preflight(printer: &Printer, env: Option<&str>, _project: Option<&PathBuf>) {
    printer.header(&format!("{} Preflight Checks", icons::SHIELD));
    
    if let Some(e) = env {
        printer.kv("Environment", e);
        printer.newline();
    }
    
    let checks = [
        ("Pipeline config", true),
        ("Git status clean", true),
        ("Secrets configured", true),
        ("SSH connection", env.is_some()),
    ];
    
    for (check, should_run) in checks {
        if !should_run {
            printer.stage(check, StageStatus::Skipped);
            continue;
        }
        
        let spin = cli::spinner(&format!("Checking {}...", check.to_lowercase()));
        std::thread::sleep(std::time::Duration::from_millis(200));
        spin.finish_and_clear();
        printer.success(check);
    }
    
    printer.newline();
    printer.success(&format!("All checks passed {}", icons::SPARKLE));
}

            .arg("-a")
            .arg("Chibby")
            .spawn()
            .ok();
    }
    
    #[cfg(target_os = "linux")]
    {
        std::process::Command::new("chibby-app")
            .spawn()
            .ok();
    }
    
    #[cfg(target_os = "windows")]
    {
        std::process::Command::new("cmd")
            .args(["/C", "start", "chibby"])
            .spawn()
            .ok();
    }
}
```

```
│  │  Output: .AppImage, .deb in target/release/   │ │
│  └───────────────────────────────────────────────┘ │
│                     ↓                               │
│  Artifacts copied to your Mac's dist/release/      │
└─────────────────────────────────────────────────────┘
```

**Example Dockerfile:**

```dockerfile
FROM rust:1.77-bookworm

    libwebkit2gtk-4.1-dev \

    libayatana-appindicator3-dev

WORKDIR /app
```

- AppImage may need extra testing

**For Windows:** Docker doesn't help - Windows requires MSVC compiler which only runs on Windows. You'd need a Windows VM or GitHub Actions for .exe builds.
